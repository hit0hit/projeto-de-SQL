CREATE TABLE diretor (
  id INTEGER PRIMARY KEY,
  nome VARCHAR(100)
);

CREATE TABLE filmes (
  id INTEGER PRIMARY KEY,
  nome varchar(100),
  ano int,
  duracao varchar(50),
  id_diretor integer,
  foreign key (id_diretor) references diretor(id)
);
